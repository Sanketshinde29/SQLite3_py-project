# connect to sqlite database
conn = sqlite3.connect("sales_data.db")
#Create a cursor to iinteract with database
cursor = conn.cursor()
#Cretae a table
cursor.execute("""
CREATE TABLE IF NOT EXISTS sales(
id INTEGER PRIMARY KEY AUTOINCREMENT,
name VARCHAR,
price INTEGER,
total_sales INTEGER
)
""")
#Commit & Close Connection
conn.commit()
